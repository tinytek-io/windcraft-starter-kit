# RadarChart

RadarChart is a component that displays data in a bar chart format.

```tsx preview
import { RadarChart } from '@/ui/charts/RadarChart';

export const data: GroupedData<"Mike" | "Lily">[] = [
  {
    name: "Math",
    Mike: 120,
    Lily: 100
  },
  {
    name: "Chinese",
    Mike: 100,
    Lily: 120
  },
  {
    name: "English",
    Mike: 110,
    Lily: 100
  },
  {
    name: "Geography",
    Mike: 120,
    Lily: 100
  },
  {
    name: "Physics",
    Mike: 100,
    Lily: 120
  },
  {
    name: "History",
    Mike: 110,
    Lily: 100
  }
];

export default function Example() {
  return (
    <RadarChart data={data} domain={[0, 150]} />
  );
}
```

```tsx preview
import { radarDataMedium } from "@/docs/charts/basic/data";
import { RadarChart } from '@/ui/charts/RadarChart';

export default function Example() {
  return (
    <RadarChart data={radarDataMedium} domain={[0, 150]} isAnimationActive={false} />
  );
}
```
