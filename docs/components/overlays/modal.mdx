# Modal

The `Modal` component is a container for a dialog that is displayed on top of the rest of the page. It is used in conjunction with the `Dialog` and `DialogTrigger` components to create a modal dialog.

```tsx
import { Heading } from '@/ui/components/Heading';
import { Dialog, DialogTrigger } from '@/ui/components/Dialog';
import { Form } from '@/ui/components/Form';
import { TextField } from '@/ui/components/TextField';
import { Modal } from '@/ui/components/Modal';
import { Button } from '@/ui/components/Button';

export default function Example() {
  return (
    <DialogTrigger>
      <Button>Sign upâ€¦</Button>
      <Modal>
        <Dialog>
          {({ close }) => (
            <Form>
              <Heading slot="title">Sign up</Heading>
              <TextField
                label="First Name"
                autoFocus
              />
              <TextField
                label="Last Name"
              />
              <Button onPress={close}>
                Submit
              </Button>
            </Form>
          )}
        </Dialog>
      </Modal>
    </DialogTrigger>
  );
}
```

## Distance to the edge

```tsx
import { Heading } from '@/ui/components/Heading';
import { Dialog, DialogTrigger } from '@/ui/components/Dialog';
import { Form } from '@/ui/components/Form';
import { TextField } from '@/ui/components/TextField';
import { Modal } from '@/ui/components/Modal';
import { Button } from '@/ui/components/Button';

export default function Example() {
  return (
    <>
      <DialogTrigger>
        <Button>Top</Button>
        <Modal position="top" isDismissable>
          <Dialog>
            {({ close }) => (<Heading slot="title">Hello</Heading>)}
          </Dialog>
        </Modal>
      </DialogTrigger>
      <DialogTrigger>
        <Button>Buttom</Button>
        <Modal position="bottom" isDismissable>
          <Dialog>
            {({ close }) => (<Heading slot="title">Hello</Heading>)}
          </Dialog>
        </Modal>
      </DialogTrigger>
      <DialogTrigger>
        <Button>Left</Button>
        <Modal position="left" isDismissable>
          <Dialog>
            {({ close }) => (<Heading slot="title">Hello</Heading>)}
          </Dialog>
        </Modal>
      </DialogTrigger>
      <DialogTrigger>
        <Button>Right</Button>
        <Modal position="right" isDismissable>
          <Dialog>
            {({ close }) => (<Heading slot="title">Hello</Heading>)}
          </Dialog>
        </Modal>
      </DialogTrigger>
      <DialogTrigger>
        <Button>Center</Button>
        <Modal position="center" isDismissable>
          <Dialog>
            {({ close }) => (<Heading slot="title">Hello</Heading>)}
          </Dialog>
        </Modal>
      </DialogTrigger>
    </>
  );
}
```
