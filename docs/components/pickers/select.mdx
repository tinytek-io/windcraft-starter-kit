# Select

The `Select` component is a form control that allows users to select an option from a list of options.

```tsx
import { Select, SelectItem } from '@/ui/components/Select';

export default function Example() {
  return (
    <Select
      label="Ice cream flavor"
      description="Select your favorite flavor"
    >
      <SelectItem>
        Chocolate
      </SelectItem>
      <SelectItem id="mint">
        Mint
      </SelectItem>
      <SelectItem>
        Strawberry
      </SelectItem>
      <SelectItem>
        Vanilla
      </SelectItem>
    </Select>
  );
}
```

## Invalid

```tsx
import { Select, SelectItem } from '@/ui/components/Select';

export default function Example() {
  return (
    <Select
      label="Ice cream flavor"
      description="Select your favorite flavor"
      isInvalid
      errorMessage="Please select a flavor"
    >
      <SelectItem>
        Chocolate
      </SelectItem>
      <SelectItem id="mint">
        Mint
      </SelectItem>
      <SelectItem>
        Strawberry
      </SelectItem>
      <SelectItem>
        Vanilla
      </SelectItem>
    </Select>
  );
}
```
